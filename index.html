create table if not exists clientes ( id bigserial primary key, nombre text not null, dni text, telefono text, email text, fecha date, tipo text default 'Activo', numero_socio text, direccion text, obs text, created_at timestamptz default now() ); create table if not exists movimientos ( id bigserial primary key, cliente_id bigint references clientes(id) on delete cascade, tipo text not null check (tipo in ('ingreso','egreso')), monto numeric(12,2) not null, fecha date not null, categoria text, detalle text, comprobante text, metodo text, usuario text, created_at timestamptz default now() ); create table if not exists usuarios ( id bigserial primary key, usuario text unique not null, pass_hash text not null, rol text default 'viewer', created_at timestamptz default now() ); create table if not exists notas ( id bigserial primary key, cliente_id bigint references clientes(id) on delete cascade, texto text not null, usuario text, created_at timestamptz default now() );
